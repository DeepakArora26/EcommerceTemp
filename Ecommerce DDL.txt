CREATE TABLE CATEGORIES
(
CATEGORY_ID VARCHAR(25) PRIMARY KEY,
CATEGORY_NAME VARCHAR(40) NOT NULL,
CATEGORY_IMAGE VARCHAR(100) NOT NULL
);



CREATE TABLE SUB_CATEGORIES
(
CATEGORY_ID VARCHAR(25) NOT NULL,
SUB_CATEGORY_ID VARCHAR(25) PRIMARY KEY,
SUBCATEGORY_NAME VARCHAR(40) NOT NULL,
SUB_CATEGORY_IMAGE VARCHAR(100) NOT NULL,
FOREIGN KEY (CATEGORY_ID) REFERENCES CATEGORIES(CATEGORY_ID)
);






CREATE TABLE PRODUCT
(
CATEGORY_ID VARCHAR(25) NOT NULL,
SUB_CATEGORY_ID VARCHAR(25) NOT NULL,
PRODUCT_ID VARCHAR(25) PRIMARY KEY,
PRODUCT_NAME VARCHAR(40) NOT NULL,
PRODUCT_MPR DOOUBLE NOT NULL,
PRODUCT_DISCOUNTED_PRICE DOUBLE NOT NULL,
PRODUCT_QUANTITY INT NOT NULL,
PRODUCT_IMAGE VARCHAR(100) NOT NULL,
PRODUCT_DESCRIPTION(500) NOT NULL,
FOREIGN KEY (SUB_CATEGORY_ID) REFERENCES SUB_CATEGORIES(SUB_CATEGORY_ID)
);


    

CREATE TABLE ORDER_HISTORY
(
ORDER_ID INT PRIMARY KEY AUTO_INCREMENT,
CUSTOMER_NAME VARCHAR(25) NOT NULL,
CART_TOTAL DOUBLE NOT NULL,
ORDER_DATE DATE NOT NULL
);





CREATE TABLE ORDER_DETAILS 
(
ORDER_ID INT,
PRODUCT_NAME VARCHAR(70) NOT NULL,
PRODUCT_QUANTITY INT NOT NULL,
PRODUCT_PRICE DOUBLE NOT NULL,
PRODUCT_TOTAL_PRICE DOUBLE NOT NULL
);
